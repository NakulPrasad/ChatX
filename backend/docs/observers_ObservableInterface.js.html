/**
 * Represents an interface for an observable in the observer pattern.
 * @interface
 */
class ObservableInterface {
    /**
     * Subscribes an observer to the observable.
     * 
     * This method should be implemented by subclasses to handle adding observers
     * to the list of subscribers.
     * 
     * @param {ObserverInterface} observer - The observer to be subscribed. Must implement the ObserverInterface.
     * @throws {Error} Throws an error if the method is not implemented by a subclass.
     */
    subscribe(observer) {
        throw new Error('Method "subscribe" must be implemented by subclasses');
    }

    /**
     * Unsubscribes an observer from the observable.
     * 
     * This method should be implemented by subclasses to handle removing observers
     * from the list of subscribers.
     * 
     * @param {ObserverInterface} observer - The observer to be unsubscribed. Must implement the ObserverInterface.
     * @throws {Error} Throws an error if the method is not implemented by a subclass.
     */
    unsubscribe(observer) {
        throw new Error('Method "unsubscribe" must be implemented by subclasses');
    }

    /**
     * Notifies all subscribed observers of a new notification.
     * 
     * This method should be implemented by subclasses to handle sending notifications
     * to all subscribed observers.
     * 
     * @param {string} notification - The notification to be sent to observers. The type of notification should be consistent with the observers' expectations.
     * @throws {Error} Throws an error if the method is not implemented by a subclass.
     */
    notify(notification) {
        throw new Error('Method "notify" must be implemented by subclasses');
    }
}

module.exports = ObservableInterface;
