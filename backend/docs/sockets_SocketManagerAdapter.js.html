/**
 * @fileoverview This module defines the SocketManagerAdapter class, which adapts the Socket.IO instance
 * to work with the SocketManagerInterface. It provides methods for handling socket events such as joining rooms,
 * disconnecting, requesting chatroom users, and sending messages.
 */

const SocketManagerInterface = require('./SocketManagerInterface.js');
const SocketIoSingleton = require('./SocketIoSingleton.js');

/**
 * @class SocketManagerAdapter
 * @extends SocketManagerInterface
 * 
 * @classdesc The SocketManagerAdapter class implements methods to manage socket events, such as handling
 * room joins, disconnections, user requests, and message sending, using a singleton instance of Socket.IO.
 */
class SocketManagerAdapter extends SocketManagerInterface {
    /**
     * Constructs the SocketManagerAdapter and initializes the Socket.IO instance.
     * 
     * @param {http.Server} httpServer - The HTTP server instance to attach the Socket.IO server to.
     */
    constructor(httpServer) {
        super();
        this.socketIoInstance = SocketIoSingleton.getInstance(httpServer);
    }

    /**
     * Handles a socket joining a room.
     * 
     * @param {Socket} socket - The socket instance representing the connected client.
     * @param {Object} data - The data provided by the client when joining a room.
     * @param {Function} cb - The callback function to be executed after the room join operation.
     */
    handleJoinRoom(socket, data, cb) {
        try {
            this.socketIoInstance.socketManager.handleJoinRoom(socket, data, cb);
        } catch (error) {
            console.error('Error handling join room:', error);
            if (cb) cb({ error: 'Internal server error while joining room' });
        }
    }

    /**
     * Handles the disconnection of a socket.
     * 
     * @param {Socket} socket - The socket instance representing the connected client.
     */
    handleDisconnect(socket) {
        try {
            this.socketIoInstance.socketManager.handleDisconnect(socket);
        } catch (error) {
            console.error('Error handling disconnect:', error);
        }
    }

    /**
     * Handles a request for chatroom users.
     * 
     * @param {Object} data - The data containing information about the chatroom.
     * @param {Function} cb - The callback function to be executed with the list of users.
     */
    handleRequestChatroomUsers(data, cb) {
        try {
            this.socketIoInstance.socketManager.handleRequestChatroomUsers(data, cb);
        } catch (error) {
            console.error('Error handling request for chatroom users:', error);
            if (cb) cb({ error: 'Internal server error while fetching chatroom users' });
        }
    }

    /**
     * Handles sending a message through the socket.
     * 
     * @param {Socket} socket - The socket instance representing the connected client.
     * @param {Object} data - The data containing the message information.
     * @param {Function} cb - The callback function to be executed after the message is sent.
     */
    handleSendMessage(socket, data, cb) {
        try {
            this.socketIoInstance.socketManager.handleSendMessage(socket, data, cb);
        } catch (error) {
            console.error('Error handling send message:', error);
            if (cb) cb({ error: 'Internal server error while sending message' });
        }
    }
}

module.exports = SocketManagerAdapter;
