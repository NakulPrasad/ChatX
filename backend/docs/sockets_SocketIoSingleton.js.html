/**
 * @fileoverview This module defines the SocketIoSingleton class, which provides a singleton instance of the
 * Socket.IO server. It ensures that only one instance of Socket.IO is created and manages socket connections
 * and events using the SocketManager.
 */

const { Server } = require('socket.io');
const SocketManager = require('./SocketManager');
const dotenv = require('dotenv');

dotenv.config();

const allowedOrigins = process.env.ORIGINS?.split('|') || [
    "http://localhost:5173",
    "https://current-bridie-nakul-384e1fc1.koyeb.app",
    "https://nakulprasad.github.io/ChatX/",
    "https://chat-x-dun.vercel.app"
];

/**
 * @class SocketIoSingleton
 * @classdesc The SocketIoSingleton class provides a singleton instance of the Socket.IO server. 
 * It ensures that only one instance of Socket.IO is created and manages socket connections and events.
 */
class SocketIoSingleton {
    /**
     * Creates an instance of SocketIoSingleton if one does not already exist.
     * 
     * @param {http.Server} httpServer - The HTTP server instance to attach the Socket.IO server to.
     * @throws {Error} Throws an error if the instance is created more than once.
     */
    constructor(httpServer) {
        if (SocketIoSingleton.instance) {
            console.warn('SocketIoSingleton instance already exists. Returning the existing instance.');
            return SocketIoSingleton.instance;
        }

        this.io = new Server(httpServer, {
            cors: {
                origin: allowedOrigins,
                methods: ["GET", "POST"]
            },
        });

        this.socketManager = new SocketManager(this.io);

        SocketIoSingleton.instance = this;
    }

    /**
     * Gets the singleton instance of SocketIoSingleton.
     * 
     * @param {http.Server} httpServer - The HTTP server instance to attach the Socket.IO server to.
     * @returns {SocketIoSingleton} The singleton instance of SocketIoSingleton.
     */
    static getInstance(httpServer) {
        if (!SocketIoSingleton.instance) {
            new SocketIoSingleton(httpServer);
        }
        return SocketIoSingleton.instance;
    }
}

module.exports = SocketIoSingleton;
